    \datavisualization [
        scientific axes={clean, standard labels},
        % school book axes,
        % style sheet=strong colors,
        % style sheet=vary dashing,
        x axis = {label={[node style={}]{\scriptsize$t$}}},
        y axis = {label={[node style={}]{\scriptsize Expectation}}},
    ]
    [
        visualize as line=expo,
        expo = {style = orange},
        expo={pin in data={text={\scriptsize Estimated $E_h[G[t]]$},when=x is 70}}
    ]
    data [format=function] {
        var x : interval [5:100];
        func y = (\value x+5) /3*(exp(-(\value x+5)/2)-1+0.5*(\value x+5))+7.9;
    }
    [
        visualize as scatter=points,
        points = {style = blue},
        points = {pin in data={text={\scriptsize$E_h[G[t]]$}, when=x is 47.5}}
    ]
    data [headline={x, y}, read from file="data/fig2_data.dat"]
    ;
